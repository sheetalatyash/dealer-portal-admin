import { Component, EventEmitter, Input, Output } from '@angular/core';
import { CommonModule } from '@angular/common';
import { MatTabChangeEvent, MatTabsModule } from '@angular/material/tabs';
import { PolarisNavigationTabConfig } from './polaris-tab-bar.class';
import { PolarisNavigationTab } from './polaris-tab-bar.types';
import { PolarisUiBase } from '../polaris-ui-base';

@Component({
    selector: 'polaris-tab-bar',
    imports: [
        CommonModule,
        MatTabsModule,
    ],
    templateUrl: './polaris-tab-bar.component.html',
    styleUrl: './polaris-tab-bar.component.scss'
})
export class PolarisTabBar extends PolarisUiBase {
  @Input() compact: boolean = false;
  @Input() stretchTabs: boolean = false;
  @Input() tabs: (PolarisNavigationTab | PolarisNavigationTabConfig)[] = [];

  @Output() tabSelected: EventEmitter<PolarisNavigationTab | PolarisNavigationTabConfig> = new EventEmitter<PolarisNavigationTab | PolarisNavigationTabConfig>();

  public get selectedIndex(): number {
    const index: number = this.tabs.findIndex((tab: PolarisNavigationTab) => tab.selected);

    return index >= 0 ? index : 0;
  }

  public emitSelectedTab(event: MatTabChangeEvent): void {
    this.tabSelected.emit(this.tabs[event.index]);
  }
}
