import { CommonModule } from '@angular/common';
import { Component, Inject, OnInit } from '@angular/core';
import { MatButtonModule } from '@angular/material/button';
import { MatIconModule } from '@angular/material/icon';
import { MatProgressBarModule } from '@angular/material/progress-bar';
import { MAT_SNACK_BAR_DATA, MatSnackBarRef } from '@angular/material/snack-bar';
import { UntilDestroy, untilDestroyed } from '@ngneat/until-destroy';
import { tap } from 'rxjs';
import { PolarisButton } from '../polaris-button';
import { PolarisIcon } from '../polaris-icon';
import { PolarisUiBase } from '../polaris-ui-base';
import { NotificationType } from './polaris-notification.enum';
import { INotificationData } from './polaris-notification.types';

@UntilDestroy()
@Component({
    selector: 'polaris-notification',
    imports: [
        CommonModule,
        MatProgressBarModule,
        MatButtonModule,
        MatIconModule,
        PolarisButton,
        PolarisIcon,
    ],
    templateUrl: './polaris-notification.component.html',
    styleUrl: './polaris-notification.component.scss'
})
export class PolarisNotification extends PolarisUiBase implements OnInit {
  public iconType: string = '';
  public progress: number = 0;
  public notificationType: string = '';
  private _currentIntervalId!: number;

  constructor(
    @Inject(MAT_SNACK_BAR_DATA) public notificationData: INotificationData,
    public snackBarRef: MatSnackBarRef<PolarisNotification>
  ) {
    super();

    this.snackBarRef.afterOpened().pipe(
      tap((): void => {
        const duration: number = this.snackBarRef.containerInstance.snackBarConfig.duration as number;
        this._runProgressBar(duration);
      }),
      untilDestroyed(this)
    ).subscribe();
  }

  public ngOnInit(): void {
    this._initializeIconClass(this.notificationData.type);
  }

  public triggerAction(): void {
    this.notificationData.onAction?.();
    this.snackBarRef.dismissWithAction();
  }

  private _runProgressBar(duration: number): void {
    const minimumProgress: number = 0;
    const maximumProgress: number = 105;
    const step: number = 0.001;

    this.progress = minimumProgress;

    this._cleanProgressBarInterval();
    this._currentIntervalId = setInterval((): void => {
      this.progress += maximumProgress * step;
      if (this.progress > maximumProgress) {
        this._cleanProgressBarInterval();
      }
    }, duration * step) as unknown as number;
  }

  private _cleanProgressBarInterval(): void {
    clearInterval(this._currentIntervalId);
  }

  private _initializeIconClass(type: NotificationType): void {
    this.notificationType = type.toLowerCase();
    this.testId = `polaris-${this.notificationType}-notification`;

    switch (type) {
      case NotificationType.Danger:
      case NotificationType.Error:
        this.iconType = 'close';
        break;
      case NotificationType.Warning:
        this.iconType = 'check';
        break;
      case NotificationType.Info:
        this.iconType = 'info';
        break;
      case NotificationType.Success:
        this.iconType = 'check';
        break;
      default:
        return;
    }
  }
}

  <div class="polaris-notification-content d-flex justify-content-between align-items-center">

      </polaris-button>

      <button

        mat-icon-button
        title="Close"
        (click)="snackBarRef.dismissWithAction()"
      >
        <polaris-icon color="dark">close-alt</polaris-icon>
      </button>
      }
    </div>
  </div>

</div>
