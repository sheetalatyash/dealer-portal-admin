import { Component, EventEmitter, Input, Output } from '@angular/core';
import { CommonModule } from '@angular/common';
import { PolarisButton } from '../polaris-button';
import { PolarisUiBase } from '../polaris-ui-base';

@Component({
    selector: 'polaris-paginator',
    imports: [
        CommonModule,
        PolarisButton,
    ],
    templateUrl: './polaris-paginator.component.html',
    styleUrl: './polaris-paginator.component.scss'
})
export class PolarisPaginator extends PolarisUiBase {
  @Input() displayedElements: number = 0;
  @Input() maxElements: number = 0;

  @Output('onLoadMore') loadMoreEmitter: EventEmitter<void> = new EventEmitter<void>();

  public onLoadMore(): void {
    this.loadMoreEmitter.emit();
  }
}

<div
  class="polaris-paginator"
  [ngClass]="customClass"
  [attr.data-test-id]="testId"
  (click)="emitClickEvent($event)"
  (mouseenter)="emitMouseEnterEvent($event)"
  (mouseleave)="emitMouseLeaveEvent($event)"
>
  <span class="polaris-paginator-count">
    Displaying {{ displayedElements > 0 ? '1' : '0' }}-{{ displayedElements }} of {{ maxElements }}
  </span>
  @if(displayedElements < maxElements) {
  <polaris-button ui-theme="secondary" ui-class="d-none d-md-block mt-2" (onClick)="onLoadMore()">
    Load More
  </polaris-button>
  <polaris-button ui-theme="tertiary" ui-class="d-md-none" (onClick)="onLoadMore()"> Load More </polaris-button>
  }
</div>
